C Fundamentals

Master the foundation of programming

🎯 11 Practice Problems

1 Print "Hello World"
Your first C program.
#include <stdio.h>

int main() {
    printf("Hello World! ");
    return 0;
}
2 Variable declaration
Declare and use different variable types.
#include <stdio.h>

int main() {
    int age = 25;
    float height = 5.9;
    char grade = 'A';
    printf("Age: %d, Height: %.1f, Grade: %c ", age, height, grade);
    return 0;
}
3 Basic arithmetic
Perform calculations with variables.
#include <stdio.h>

int main() {
    int a = 10, b = 3;
    printf("Add: %d ", a + b);
    printf("Subtract: %d ", a - b);
    printf("Multiply: %d ", a * b);
    printf("Divide: %d ", a / b);
    return 0;
}
4 If statement
Use conditional statements.
#include <stdio.h>

int main() {
    int age = 20;
    if(age >= 18) {
        printf("Adult ");
    } else {
        printf("Minor ");
    }
    return 0;
}
5 For loop 1-10
Print numbers 1 to 10 using for loop.
#include <stdio.h>

int main() {
    for(int i = 1; i <= 10; i++) {
        printf("%d ", i);
    }
    printf(" ");
    return 0;
}
6 Simple function
Create and call a custom function.
#include <stdio.h>

void sayHello(char name[]) {
    printf("Hello %s! ", name);
}

int main() {
    sayHello("John");
    return 0;
}
7 Array declaration
Create and display an array.
#include <stdio.h>

int main() {
    int numbers[5] = {10, 20, 30, 40, 50};
    for(int i = 0; i < 5; i++) {
        printf("%d ", numbers[i]);
    }
    return 0;
}
8 Basic scanf/printf
Get input from user and display it.
#include <stdio.h>

int main() {
    int age;
    printf("Enter your age: ");     scanf("%d", &age);
    printf("You are %d years old ", age);
    return 0;
}
9 Simple switch
Use switch statement for menu choices.
#include <stdio.h>

int main() {
    int choice = 2;
    switch(choice) {
        case 1: printf("Add "); break;
        case 2: printf("Subtract "); break;
        default: printf("Other ");
    }
    return 0;
}
10 Basic pointer
Declare and use a pointer.
#include <stdio.h>

int main() {
    int num = 42;
    int *ptr = &num;
    printf("Value: %d ", num);
    printf("Address: %p ", (void*)ptr);
    printf("Value via ptr: %d ", *ptr);
    return 0;
}
11 🎉 COMBO CHALLENGE: Complete C Program
Combine ALL 10 C skills into one complete program!
#include <stdio.h>

void printNumbers() {
    for(int i = 1; i <= 10; i++) {
        printf("%d ", i);
    }
    printf(" ");
}

int main() {
    printf("=== C Programming Demo === ");

    // Variables
    int age;
    printf("Enter age: ");     scanf("%d", &age);

    // If statement
    if(age >= 18) {
        printf("Adult ");
    } else {
        printf("Minor ");
    }

    // Function call
    printNumbers();

    // Array
    int nums[5] = {10, 20, 30, 40, 50};
    for(int i = 0; i < 5; i++) {
        printf("%d ", nums[i]);
    }
    printf(" ");

    // Pointer
    int *ptr = &nums[0];
    printf("First via pointer: %d ", *ptr);

    // Switch
    int choice;
    printf("1=Add 2=Subtract: ");     scanf("%d", &choice);
    switch(choice) {
        case 1: printf("Addition "); break;
        case 2: printf("Subtraction "); break;
        default: printf("Other ");
    }

    printf("✅ Complete! ");
    return 0;
}
