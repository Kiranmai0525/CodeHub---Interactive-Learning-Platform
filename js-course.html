JavaScript Fundamentals

Make your websites interactive and alive

🎯 11 Practice Problems

1 Show alert box
Display a popup message to users.
<!DOCTYPE html>
<html>
<head>
<title>Alert</title>
</head>
<body>
<script>
alert("Hello World!");
</script>
</body>
</html>
2 Change text content
Update text inside an HTML element.
<p id="demo">Original text</p>
<script>
document.getElementById("demo").innerHTML = "New text!";
</script>
3 Get button click
Detect when user clicks a button.
<button id="myBtn">Click Me</button>
<script>
document.getElementById("myBtn").onclick = function() {
alert("Button clicked!");
}
</script>
4 Console.log message
Print message to browser console (F12 to see).
<script>
console.log("Hello from console!");
console.log("This appears in DevTools");
</script>
5 Create variable
Store data in a variable.
<script>
let name = "John";
let age = 25;
alert("Name: " + name + ", Age: " + age);
</script>
6 Simple if statement
Make decisions with if condition.
<script>
let age = 18;
if (age >= 18) {
alert("You are an adult!");
} else {
alert("You are a minor!");
}
</script>
7 Basic function
Create reusable code with functions.
<script>
function sayHello(name) {
alert("Hello " + name + "!");
}
sayHello("John");
</script>
8 Change background color
Change webpage background dynamically.
<button onclick="changeColor()">Change Color</button>
<script>
function changeColor() {
document.body.style.backgroundColor = "#FF6B6B";
}
</script>
9 Get input value
Read user input from form field.
<input type="text" id="userInput" placeholder="Enter name">
<button onclick="showInput()">Show Input</button>
<p id="output"></p>
<script>
function showInput() {
let value = document.getElementById("userInput").value;
document.getElementById("output").innerHTML = "You typed: " + value;
}
</script>
10 Simple counter
Create a click counter.
<p>Count: <span id="counter">0</span></p>
<button onclick="increment()">+1</button>
<script>
let count = 0;
function increment() {
count++;
document.getElementById("counter").innerHTML = count;
}
</script>
11 🎉 COMBO CHALLENGE: Interactive Portfolio
Combine ALL 10 JS skills into one interactive webpage!
<!DOCTYPE html>
<html>
<head>
<title>Interactive Portfolio</title>
<style>
body { font-family: Arial; text-align: center; padding: 20px; }
button { padding: 10px 20px; font-size: 16px; margin: 5px; }
#counter { font-size: 24px; font-weight: bold; color: #FF6B6B; }
</style>
</head>
<body>
<h1 id="title">John Doe</h1>
<p>Full Stack Developer</p>
<input type="text" id="nameInput" placeholder="Your name">
<br><br>
<button onclick="greetUser()">Greet Me</button>
<button onclick="changeTheme()">Change Theme</button>
<button onclick="showAge()">Check Age</button>
<p>Clicks: <span id="clickCounter">0</span></p>
<p id="output"></p>

<script>
let clickCount = 0;

function greetUser() {
let name = document.getElementById("nameInput").value || "Friend";
alert("Hello " + name + "! Welcome to my portfolio!");
console.log("User greeted:", name);
}

function changeTheme() {
let colors = ["#FF6B6B", "#4ECDC4", "#FFE66D"];
let randomColor = colors[Math.floor(Math.random() * 3)];
document.body.style.backgroundColor = randomColor;
}

function showAge() {
let age = 25;
if (age >= 18) {
document.getElementById("output").innerHTML = "✅ Adult: " + age + " years old";
} else {
document.getElementById("output").innerHTML = "👶 Minor";
}
}

function incrementCounter() {
clickCount++;
document.getElementById("clickCounter").innerHTML = clickCount;
document.getElementById("title").innerHTML = "John Doe - " + clickCount + " clicks";
}

// Add click counter to all buttons
greetUser = incrementCounter + greetUser;
changeTheme = incrementCounter + changeTheme;
showAge = incrementCounter + showAge;
</script>
</body>
</html>
